          </main>
      )}
      {level === 7 && (
          <main className="flex-1 overflow-y-auto bg-white dark:bg-slate-900 p-6">
              <CombinationRepetitionViz />
          </main>
      {level === 8 && (
          <main className="flex-1 overflow-y-auto bg-white dark:bg-slate-900 p-6">
              <CircularPermutationViz />
          </main>
      )}
      )}
    </div>
  );
}
